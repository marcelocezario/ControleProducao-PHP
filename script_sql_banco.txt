/************* Gerar Tabelas Banco *************\

CREATE TABLE estado (

id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
sigla VARCHAR(2) NOT NULL,
descricao VARCHAR(30) NOT NULL

);

CREATE TABLE cliente (

id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
acesso decimal(1),
apelido varchar(80),
ativo boolean,
bairro varchar(80),
cep varchar(10),
cidade varchar (80),
complemento varchar(80),
cpf varchar(17),
dtNascimento varchar(11),
email varchar(80),
idEstado int,
logradouro varchar(80),
nome varchar(80),
numero varchar(6),
senha varchar(20),
telefone varchar(15),
FOREIGN KEY (idEstado) REFERENCES estado (id)
);

create table categoria(

id int not null PRIMARY KEY AUTO_INCREMENT,
descricao varchar(150),
nomeCategoria varchar(80)

);

create table marca(

id int not null PRIMARY KEY AUTO_INCREMENT,
nomeMarca varchar(80),
fornecedor varchar(150)

);

create table produto(

id int not null PRIMARY KEY AUTO_INCREMENT,
ativo boolean,
descricaoResumida varchar(200),
descricaoCompleta varchar(2000),
idCategoria INTEGER,
idMarca INTEGER,
nomeProduto varchar(80),
qtdeEstoque INTEGER,
url varchar(200),
valor decimal(10,2),
FOREIGN KEY (idCategoria) REFERENCES categoria (id),
FOREIGN KEY (idMarca) REFERENCES marca (id)

);

create table meioPagamento(

id int not null PRIMARY KEY AUTO_INCREMENT,
formaPagamento varchar (80),
nrMaxParcelas integer,
txJurosParcelamento decimal(10,2)

);

create table venda(

id int not null PRIMARY KEY AUTO_INCREMENT,
bairro varchar(80),
cep varchar(10),
cidade varchar (80),
complemento varchar(80),
data date,
idCliente integer,
idEstado int,
idMeioPagamento integer,
logradouro varchar(80),
nomeCliente varchar (80),
nrParcelas integer,
nrPedido varchar(80),
numero varchar(6),
valorFrete decimal(10,2),
valorTotal decimal(10,2),
valorParcela decimal (10,2),
FOREIGN KEY (idCliente) REFERENCES cliente (id),
FOREIGN KEY (idEstado) REFERENCES estado (id),
FOREIGN KEY (idMeioPagamento) REFERENCES meiopagamento (id)

);


create table itemVenda(	

id int not null PRIMARY KEY AUTO_INCREMENT,
idProduto INTEGER,
idVenda INTEGER,
nomeProduto varchar (80),
valorProduto decimal(10,2),
qtde INTEGER,
valorTotal decimal(10,2),
foreign key (idProduto) references produto (id),
foreign key (idVenda) references venda (id)

);
