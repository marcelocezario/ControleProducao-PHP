/************* Categorias *************\
create table categoria(
	id int not null PRIMARY KEY AUTO_INCREMENT,
    nomeCategoria varchar(80),
    descricao varchar(150)
);
/************* Produtos *************\
create table produtos(
	id int not null PRIMARY KEY AUTO_INCREMENT,
    qtde decimal(10,2),
    id_categoria int
);
ALTER TABLE `produtos` ADD CONSTRAINT `id_categoria` FOREIGN KEY ( `id_categoria` ) REFERENCES `categoria` ( `id` ) ;

/************* Cadastro de cliente *************\

CREATE TABLE cliente (
    id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
    nome varchar(80),
    dtNascimento varchar(11),
    cpf varchar(17),
    telefone varchar(15),
    cep varchar(10),
    rua varchar(80),
    numero decimal,
    bairro varchar(80),
    id_estado decimal,
    email varchar(80),
    senha varchar(20),
    ativo boolean,
    acesso decimal(1)
);

/************* estado *************\

CREATE TABLE estados (
  id INTEGER NOT NULL PRIMARY KEY AUTO_INCREMENT,
  sigla VARCHAR(2) NOT NULL,
  descricao VARCHAR(30) NOT NULL
);
